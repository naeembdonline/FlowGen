--------------------------------------------------------------------------------
📊 Common Issues & Fixes
--------------------------------------------------------------------------------

ISSUE: "unzip: command not found"
FIX: apt install unzip -y

ISSUE: "Docker not found"
FIX: curl -fsSL https://get.docker.com | sh

ISSUE: "Permission denied"
FIX: Ensure you're logged in as root

ISSUE: "Port already in use"
FIX: docker stop $(docker ps -q) && systemctl stop nginx
